import { createForm } from "@createform/react";

const useContactForm = createForm({
  initialValues: {
    name: "",
    email: "",
    message: "",
  },
  mode: "onChange",
});

  const { register, handleSubmit, state } = useContactForm();

    alert(`Form submitted! Name: ${data.name}, Email: ${data.email}`);

  return (

              {errors.name}

**What you're seeing:**

- **`createForm`**: Creates a form hook with initial values
- **`register`**: Connects form fields to the form state
- **`handleSubmit`**: Handles form submission with validation
- **`state`**: Contains current form state (errors, touched, isValid)

**Key concepts:**
- Forms start with `initialValues` 
- The `mode` controls when validation runs
- Field registration is automatic with `register()`
- Form state updates in real-time

import { createForm } from "@createform/react";

// Custom validation functions
const customValidations = {
  required: (value, message = "This field is required") => {
    if (!value || value.toString().trim() === "") {
      return message;
    }
    return null;
  },
  
  minLength: (min) => (value, message) => {
    if (value && value.length < min) {
      return message || `Must be at least ${min} characters`;
    }
    return null;
  },
  
  isValidEmail: (value, message = "Please enter a valid email") => {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (value && !emailRegex.test(value)) {
      return message;
    }
    return null;
  },
  
  matchesField: (fieldToMatch) => (value, message, formValues) => {
    if (value && value !== formValues[fieldToMatch]) {
      return message || `This field must match ${fieldToMatch}`;
    }
    return null;
  },
  
  isStrongPassword: (value, message = "Password is not strong enough") => {
    if (!value) return null;
    
    const hasUpper = /[A-Z]/.test(value);
    const hasLower = /[a-z]/.test(value);
    const hasNumber = /\d/.test(value);
    const hasSpecial = /[!@#$%^&*]/.test(value);
    const isLongEnough = value.length >= 8;
    
    if (!hasUpper || !hasLower || !hasNumber || !hasSpecial || !isLongEnough) {
      return message;
    }
    
    return null;
  }
};

// Custom validation schema
const customValidationSchema = (values) => {
  const errors = {};
  
  // Username validation
  const usernameError = 
    customValidations.required(values.username, "Username is required") ||
    customValidations.minLength(3)(values.username, "Username must be at least 3 characters");
  if (usernameError) errors.username = usernameError;
  
  // Email validation
  const emailError = 
    customValidations.required(values.email, "Email is required") ||
    customValidations.isValidEmail(values.email);
  if (emailError) errors.email = emailError;
  
  // Password validation
  const passwordError = 
    customValidations.required(values.password, "Password is required") ||
    customValidations.isStrongPassword(values.password);
  if (passwordError) errors.password = passwordError;
  
  // Confirm password validation
  const confirmPasswordError = 
    customValidations.required(values.confirmPassword, "Please confirm your password") ||
    customValidations.matchesField("password")(values.confirmPassword, "Passwords don't match", values);
  if (confirmPasswordError) errors.confirmPassword = confirmPasswordError;
  
  return errors;
};

import { createForm } from "@createform/react";
import { useState } from "react";

const useDynamicForm = createForm({
  initialValues: {
    userType: "",
    // Student fields
    studentId: "",
    school: "",
    graduationYear: "",
    // Professional fields
    company: "",
    jobTitle: "",
    experience: "",
    // General fields
    name: "",
    email: "",
    subscribe: false,
  },
  mode: "onChange",
});

export default function ConditionalFieldsForm() {
  const { register, handleSubmit, state, setFieldValue } = useDynamicForm();
  const { values, errors, touched } = state;

  const handleUserTypeChange = (type) => {
    setFieldValue("userType", type);
    
    // Clear other fields when switching user type
    if (type === "student") {
      setFieldValue("company", "");
      setFieldValue("jobTitle", "");
      setFieldValue("experience", "");
    } else if (type === "professional") {
      setFieldValue("studentId", "");
      setFieldValue("school", "");
      setFieldValue("graduationYear", "");
    }
  };

  const onSubmit = (data) => {
    console.log("Dynamic form submitted:", data);
    alert(`Registration successful as ${data.userType}!`);
  };

  return (
    <div style={{ maxWidth: '500px', margin: '0 auto' }}>
      <form onSubmit={handleSubmit(onSubmit)}>
        {/* User Type Selection */}
        <div style={{ marginBottom: '2rem' }}>
          <label style={{ display: 'block', marginBottom: '8px', fontWeight: 'bold' }}>
            I am a:
          </label>
          <div style={{ display: 'flex', gap: '1rem' }}>
            <label style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
              <input 
                type="radio"
                name="userType"
                value="student"
                checked={values.userType === "student"}

                cursor: isStepValid(currentStep) ? 'pointer' : 'not-allowed'
              }}
            >
              Submit
            </button>
          )}
        </div>
      </form>
    </div>
  );
